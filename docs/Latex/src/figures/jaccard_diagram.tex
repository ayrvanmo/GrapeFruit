\node (output) [startstop, below of=calc_jaccard] {Mostrar el resultado};

% Flechas del algoritmo de Jaccard
\draw [arrow] (start) -- (calc_inter);
\draw [arrow] (calc_inter) -- (calc_union);
\draw [arrow] (calc_union) -- (calc_jaccard);
\draw [arrow] (calc_jaccard) -- (output);
